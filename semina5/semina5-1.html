import cv2
import numpy as np
from matplotlib import pyplot as plt

original_image = cv2.imread("../data/mean-shift-data.png")
img = cv2.cvtColor(original_image,cv2.COLOR_BGR2RGB)# rgb   
vectorized = img.reshape((-1,3))
#1 
vectorized = np.float32((vectorized))

criteria = (cv2.TERM_CRITERIA_EPS + cv2.TERM_CRITERIA_MAX_ITER,10,1.0)
attempts = 10


fig =plt.figure(figsize=(11,11))
plt.subplot(321),plt.title("original")# 3 2 1
plt.imshow(img)

count=1
for k in [3,4,5,6,7]:
    count=count+1
    ret,label,center = cv2.kmeans(vectorized,
                                  k,
                                  None,
                                  criteria,attempts,
                                  cv2.KMEANS_PP_CENTERS)
    #KMEANS_PP_CENTERS : kmeans++  
    print("center ",center)
    center = np.uint8(center)
    res= center[label.flatten()] #center     .
    result_image=res.reshape((img.shape))
    plt.subplot(320+count),plt.title("K="+str(k))# 3 2 1
    plt.imshow(result_image)
    
plt.show()
