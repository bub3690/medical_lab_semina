import numpy as np
import pandas as pd
from matplotlib import pyplot as plt


DATA_COV=np.cov(DATA_norm.T)
print('eigenvector : ',DATA_COV.shape[1])

#decomposition , eigen value    .
## ,    .
eig_vals, eig_vecs=np.linalg.eig(DATA_COV)
print("  : ", np.sum(eig_vals[0:2])/np.sum(eig_vals))
U=eig_vecs[:,0:2]

# 
DATA_PCA=DATA_norm.dot(U)


DATA_PCA.columns=['PCA1','PCA2']

fig =plt.figure(figsize=(8,6))
benign= DATA_PCA[raw_data['target']==1] #
malignant = DATA_PCA[raw_data['target']==0] #

plt.scatter(benign.iloc[:,0],benign.iloc[:,1],label='')
plt.scatter(malignant.iloc[:,0],malignant.iloc[:,1],label='')

plt.xlabel('PCA FIRST COMPONENT')
plt.ylabel('PCA SECOND COMPONENT')
plt.legend()
plt.show()

# pca  .
from sklearn.preprocessing import MinMaxScaler
from sklearn.decomposition import PCA

df2 = pd.DataFrame(raw_data['data'],columns=raw_data['feature_names'])
scaler=MinMaxScaler()
scaler.fit(df2)
scaled_data=scaler.transform(df2)
pca2=PCA(n_components=2)
pca2.fit(scaled_data)

pca2_data=pca2.transform(scaled_data)
pca2_df=pd.DataFrame(pca2_data,columns=['PCA1','PCA2'])

fig =plt.figure(figsize=(8,6))
benign2= pca2_df[raw_data['target']==1] #
malignant2 = pca2_df[raw_data['target']==0] #

plt.scatter(benign2.iloc[:,0],benign2.iloc[:,1],label='')
plt.scatter(malignant2.iloc[:,0],malignant2.iloc[:,1],label='')

plt.xlabel('PCA FIRST COMPONENT')
plt.ylabel('PCA SECOND COMPONENT')
plt.legend()
plt.show()
pca2_components_df=pd.DataFrame(pca2.components_,columns=raw_data['feature_names'],index=['pca1','pca2'])

fig =plt.figure(figsize=(10,6))
ax = sns.heatmap(pca2_components_df,
                 cmap='RdYlGn_r',# cmap Color
                 annot=False,            # Value Text
                 square=True,
                )
