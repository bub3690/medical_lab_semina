# 3. MLP  . hidden layer 2??
class Net(nn.Module):
    def __init__(self):
        super(Net, self).__init__() # nn   .
        self.fc1 = nn.Linear(28*28, 512) # 1. input . 28*28. 512 output .
        self.fc2 = nn.Linear(512,256) #2. hidden 1,  512 node.fully connected.
        self.fc3 = nn.Linear(256,10) #3. hidden 2.
        # 10 output 
    def forward(self,x):
        x = x.view(-1,28*28) # x 2 . 1 FLATTEN
        x = self.fc1(x) # input  . 512 out
        x = F.sigmoid(x) #   sigmoid->hidden layer
        x = self.fc2(x) # hidden  . 256 out
        x = F.sigmoid(x) #   sigmoid
        x = self.fc3(x) # hidden  . 10 out
        x = F.log_softmax(x,dim = 1)# softmax  0~10 .
        return x

#6.   validation 
def evaluate(model,test_loader):
    model.eval()# 
    test_loss =0
    correct =0
    
    #no_grad    , gradient x
    with torch.no_grad():
        for image,label in test_loader:
            image = image.to(DEVICE)
            label = label.to(DEVICE)
            output = model(image)
            test_loss +=criterion(output,label).item()#batch  loss .
            prediction = output.max(1,keepdim=True)[1]#10 
            correct +=prediction.eq(label.view_as(prediction)).sum().item()#  .
            #view_as    .
    test_loss /= len(test_loader.dataset)#batch loss 
    test_accuracy = 100. * correct/len(test_loader.dataset)
    return test_loss, test_accuracy
