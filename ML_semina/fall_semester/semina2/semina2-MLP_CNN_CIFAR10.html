def train(model,train_loader,optimizer, log_interval):
    model.train()
    for batch_idx,(image,label) in enumerate(train_loader):
        image = image.to(DEVICE)
        label = label.to(DEVICE)
        #  
        optimizer.zero_grad() # device   gradient 
        output = model(image) # model output 
        loss = criterion(output, label) #loss 
        loss.backward() # loss   gradient 
        optimizer.step() # Gradient    .

#9.  , test   
def evaluate(model,test_loader):
    model.eval()
    test_loss = 0
    correct = 0
    #no_grad :    .
    with torch.no_grad():
        for image, label in test_loader:
            image = image.to(DEVICE)
            label = label.to(DEVICE)
            output = model(image)
            test_loss += criterion(output, label).item()
            prediction = output.max(1,keepdim=True)[1] #    class 1 . 
            #print(output.max(1,keepdim=True))
            #print(prediction)
            correct += prediction.eq(label.view_as(prediction)).sum().item()#    32.
            #true.false sum. item
        test_loss /= len(test_loader.dataset)
        test_accuracy = 100. * correct / len(test_loader.dataset)
        return test_loss,test_accuracy

# 6-2. . drop out .
class Net(nn.Module):
    def __init__(self):
        super(Net,self).__init__() # 
        #     
        self.conv1 = nn.Conv2d(
            in_channels = 3,# RGB,  
            out_channels = 8,# FeatureMap , 
            kernel_size = 3,#3x3
            padding=1,#. 
        )
        # 3  -> 8 .
        self.conv2 = nn.Conv2d(
            in_channels = 8,# featureMap 8
            out_channels = 16,# FeatureMap , 
            kernel_size = 3,#3x3
            padding=1,#. 
        )
        #8  -> 16 
        self.pool = nn.MaxPool2d(
            kernel_size = 2,
            stride = 2
        )
        #  4 1.
        
        #Fully-connected layer
        self.fc1 = nn.Linear(8*8*16,64)# pooling 8*8 
        self.fc2 = nn.Linear(64,32)
        self.fc3 = nn.Linear(32,10)
        
    
    def forward(self,x):
        x = self.conv1(x)#flatten  .
        x = F.relu(x)
        x = self.pool(x)
        
        x = self.conv2(x)
        x = F.relu(x)
        x = self.pool(x)
        
        x = x.view(-1,8*8*16)# flatten
        x = self.fc1(x)
        x = F.relu(x)

        x = self.fc2(x)
        x = F.relu(x)
        
        x = self.fc3(x)
        x = F.log_softmax(x,dim=1)#10 
        return x
